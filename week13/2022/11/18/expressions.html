Homework/Binary Adaptation: Create a python function that will convert a decimal number 1-255 to binary using mathematical operations and powers of 2.

def convert(dec):
    bin = ""
    i = 7

    while i >= 0:
        if dec % (2**i) == dec:
            bin = bin + "0"
            i -= 1
        else:
            bin = bin + "1"
            dec -= 2**i
            i -= 1

    print(bin)
